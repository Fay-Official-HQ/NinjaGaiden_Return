# 《忍者龙剑传：归来》— 灭杀系统开发总结

> 生成日期：2026-07-06
> 引擎：Godot 4.6
> 项目路径：`E:\Godot\GodotPro\NinjaGaiden_Return`

---

## 一、需求概述

实现一个类《忍者龙剑传》的**灭杀系统**，使用独立按键 **H 键**触发，与 J 键普通攻击完全分离。

### 核心机制

1. **蓄力攻击**：按住 H 键 ≥0.3s 进入蓄力状态，身体渐红，每 0.5s 积累 1 层能量（≤6 层），松开后释放强力斩击（伤害 = 1 + 层数）。蓄力期间玩家可自由移动、跳跃、下蹲。
2. **灭杀连斩**：第一次蓄力攻击斩杀敌人后，0.5s 内按 H 键，自动在 ExterminateDetector 范围内搜寻最近敌人 → 瞬移到其身旁 → 播放灭杀动画 → 造成 1 点伤害 → 刷新 0.5s 窗口 → 继续（≤6 次）。
3. **三种姿态动画**：
   - `Exec_Stand` — 站立/跑动时释放
   - `Exec_Crouch` — 下蹲时释放
   - `Exec_Air` — 跳跃/坠落时释放

**函数 `_update_charge(delta)`** — 蓄力逻辑：

```gdscript
func _update_charge(delta: float) -> void:
    # 按下 H → 初始化蓄力
    if Input.is_action_just_pressed("exterminate"):
        _charge_hold_time = 0.0
        exterminate_stacks = 0
        _charge_energy_timer = 0.0
        _charge_visual_active = false

    # 按住 H → 累积时间
    if Input.is_action_pressed("exterminate") and _charge_hold_time >= 0:
        _charge_hold_time += delta
        if _charge_hold_time >= 0.3:
            if not _charge_visual_active:
                _charge_visual_active = true
                AudioManager.play_sound(&"renshuhuoqiu")
            _charge_energy_timer += delta
            if _charge_energy_timer >= 0.5 and exterminate_stacks < 6:
                exterminate_stacks += 1
                _charge_energy_timer -= 0.5
            # 身体渐红（0~3秒内红色饱和度 0~80%）
            var redness = min((_charge_hold_time - 0.3) / 2.7, 1.0)
            animated_sprite.modulate = Color(1.0, 1.0 - redness * 0.8, 1.0 - redness * 0.8)

    # 松开 H → 释放或取消
    if Input.is_action_just_released("exterminate"):
        if _charge_hold_time >= 0.3:
            var current = state_machine.current_state
            var anim_name = "Exec_Stand"
            if current is JumpState or current is FallState:
                anim_name = "Exec_Air"
            elif current is CrouchState:
                anim_name = "Exec_Crouch"
            state_machine.change_state(
                $StateMachine/ExterminateReleaseState,
                {"energy": exterminate_stacks, "anim_name": anim_name}
            )
        else:
            _cancel_charge()

    # H 完全松开 → 清理
    if not Input.is_action_pressed("exterminate") and not Input.is_action_just_pressed("exterminate"):
        if _charge_hold_time >= 0:
            _cancel_charge()
```

### 3.2 `ExterminateReleaseState.gd`（新建）

**路径**：`scripts/player/states/ExterminateReleaseState.gd`

完整代码（约 79 行）：

```gdscript
extends State
class_name ExterminateReleaseState

var _energy: int = 0
var _killed_any: bool = false
var _hit_enemies: Array[HurtBox] = []
var _attack_root: Node2D
var _dash_box: SwordHitBox
var _release_timer: float = 0.0
var _execution_duration: float = 0.0
var _anim_name: String = "Exec_Air"

func enter(msg: Dictionary = {}) -> void:
    _energy = msg.get("energy", 0)
    _anim_name = msg.get("anim_name", "Exec_Air")
    _killed_any = false
    _hit_enemies.clear()
    _release_timer = 0.0

    # 根据动画名动态计算时长
    var sprite = player.animation.sprite
    var frame_count = sprite.sprite_frames.get_frame_count(_anim_name)
    var speed = sprite.sprite_frames.get_animation_speed(_anim_name)
    _execution_duration = frame_count / speed

    player.animation.play(_anim_name)
    AudioManager.play_sound(&"bishaji")

    _attack_root = player.get_node("AttackRoot") as Node2D
    _dash_box = _attack_root.get_node("SwordDashBox") as SwordHitBox
    if _attack_root:
        _attack_root.scale.x = player.facing_direction

func update(delta: float) -> void:
    _release_timer += delta

    # 动画 33% 处启用 SwordDashBox 攻击框
    if _dash_box and not _dash_box.monitoring and _release_timer >= _execution_duration * 0.33:
        _dash_box.set_deferred("monitoring", true)

    # 轮询检测 HurtBox → 造成伤害
    if _dash_box and _dash_box.monitoring:
        var areas = _dash_box.get_overlapping_areas()
        for area in areas:
            if area is HurtBox and not _hit_enemies.has(area):
                _hit_enemies.append(area)
                var damage = 1 + _energy
                area.take_damage(damage)
                var enemy = area.owner
                if is_instance_valid(enemy):
                    var dead = ("is_dead" in enemy and enemy.is_dead) or ("_is_dead" in enemy and enemy._is_dead)
                    if dead:
                        _killed_any = true

    # 动画结束 → 开启链式窗口 / 结束
    if _release_timer >= _execution_duration:
        _cleanup()
        if _killed_any and _energy > 0:
            player.exterminate_remaining_chains = _energy
            player.exterminate_chain_timer = 0.5
            player.exterminate_chain_active = true
        if player.is_on_floor():
            state_machine.change_state(player.idle_state)
        else:
            state_machine.change_state(player.fall_state)
        return

func physics_update(_delta: float) -> void:
    player.velocity.x = 0.0
    player.velocity.y = 0.0
    player.move_and_slide()

func exit() -> void:
    _cleanup()

func _cleanup() -> void:
    if _dash_box:
        _dash_box.set_deferred("monitoring", false)
    _hit_enemies.clear()
    if _attack_root:
        _attack_root.scale.x = 1.0
    if not _killed_any or _energy <= 0:
        player._end_exterminate_chain()
```

### 3.3 `ExterminateChainState.gd`（新建）

**路径**：`scripts/player/states/ExterminateChainState.gd`

完整代码（约 77 行）：

```gdscript
extends State
class_name ExterminateChainState

var _target: Node2D
var _chains_left: int = 0
var _execution_duration: float = 0.0
var _timer: float = 0.0
var _has_dealt_damage: bool = false

func enter(msg: Dictionary = {}) -> void:
    _target = msg.get("target", null)
    _chains_left = msg.get("chains", 0) - 1

    if not is_instance_valid(_target) or _is_node_dead(_target):
        player._end_exterminate_chain()
        state_machine.change_state(_recover_state())
        return

    var sprite = player.animation.sprite
    var frame_count = sprite.sprite_frames.get_frame_count("Exec_Air")
    var speed = sprite.sprite_frames.get_animation_speed("Exec_Air")
    _execution_duration = frame_count / speed
    _timer = 0.0
    _has_dealt_damage = false

    # 瞬移到目标附近
    var offset_x = 20.0 * (-1.0 if _target.global_position.x > player.global_position.x else 1.0)
    player.global_position = _target.global_position + Vector2(offset_x, -8.0)
    player.set_facing_direction(1.0 if _target.global_position.x > player.global_position.x else -1.0)

    player.animation.play("Exec_Air")
    AudioManager.play_sound(&"gongji")
    player.is_invincible = true

func update(delta: float) -> void:
    _timer += delta

    # 动画 33% 处造成 1 点伤害
    if not _has_dealt_damage and _timer >= _execution_duration * 0.33:
        _has_dealt_damage = true
        if is_instance_valid(_target) and not _is_node_dead(_target):
            var hurtbox = _target.get_node_or_null("HurtBox") if "hurtbox" not in _target else _target.hurtbox
            if hurtbox is HurtBox:
                hurtbox.take_damage(1)

    # 动画结束 → 判断是否继续链式
    if _timer >= _execution_duration:
        var killed = not is_instance_valid(_target) or _is_node_dead(_target)
        if killed and _chains_left > 0:
            player.exterminate_remaining_chains = _chains_left
            player.exterminate_chain_timer = 0.5
            player.exterminate_chain_active = true
        else:
            player._end_exterminate_chain()
        state_machine.change_state(_recover_state())

func physics_update(_delta: float) -> void:
    player.velocity = Vector2.ZERO
    player.move_and_slide()

func exit() -> void:
    player.is_invincible = false

func _recover_state() -> State:
    if player.is_on_floor():
        return player.idle_state
    else:
        return player.fall_state

func _is_node_dead(node: Node2D) -> bool:
    if not is_instance_valid(node):
        return true
    if "is_dead" in node and node.is_dead:
        return true
    if "_is_dead" in node and node._is_dead:
        return true
    return false
```

## 四、蓄力系统设计要点

### 4.1 为何不使用独立状态

蓄力**不在状态机中**作为一个独立状态，而是在 `Player._process()` 中全局跟踪。这样设计的目的是：

- 蓄力时不锁死玩家，可以**自由移动、跳跃、下蹲**
- 在任何合法状态下松开 H 都能释放
- 受伤时自动取消蓄力

### 4.2 关键设计决策

| 决策 | 说明 |
|:-----|:------|
| 使用独立 H 键 | 与 J 键普攻完全分离，不干扰普攻手感 |
| `_process` 跟踪蓄力 | 不阻塞移动/跳跃，蓄力时玩家行为完全自由 |
| 空中可蓄力释放 | 跳跃/坠落状态松开 H 播放 `Exec_Air`，浮动执行动画 |
| 身体渐红 | 0.3s~3.0s 内 Color 从白色渐变到红色饱和度 80% |
| 0.5s 链式窗口 | 斩杀后刷新 0.5s 计时，按 H 触发链式，超时变回白色 |
| 链式瞬移 | `ExterminateChainState.enter()` 中直接设置 `player.global_position` 到目标附近 |
| 无敌 | 执行灭杀动画时 `player.is_invincible = true`，避免中途受伤 |

### 4.3 能量与伤害

- 蓄力 ≥0.3s 进入蓄力状态
- 每多按住 0.5s 积累 1 层（蓄力音效在进入时播放一次）
- 最多 6 层
- 释放伤害 = 1 + 层数
- 链式斩每次固定 1 点伤害

### 5.2 敌人搜索逻辑

## 七、已知注意事项

1. **`_is_node_dead()` 辅助方法**：需同时存在于 `Player.gd` 和 `ExterminateChainState.gd`（均有调用）
2. **`Object.get("prop")` vs `Dictionary.get("key", default)`**：`Object.get()` 只能传 1 个参数，用于属性名。默认值语法只在 `Dictionary.get()` 中有效
3. **`"prop" in obj` 语法**：GDScript 中检测属性/成员是否存在，兼容 `BaseEnemy.is_dead` 和 `EagleEnemy._is_dead` 两种命名
4. **`execution` 动画已删除**：旧名 `"execution"` 的动画已被 `Exec_Stand` / `Exec_Crouch` / `Exec_Air` 替代，代码中不要引用 `"execution"`
5. **纹理引用问题**：如果在新电脑上纹理丢失，需要在 SpriteFrames 编辑器中手动重新指定 `Exec_Stand` / `Exec_Crouch` / `Exec_Air` 各帧引用的纹理图片
